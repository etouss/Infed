\section{Semantics}

\begin{mydef}
$$ \sigma_{\Sigma}(x) = (x[r_i.a_i] | r_i.a_i \in \Sigma) $$
$$ \sigma_{*}(x) = x $$
\end{mydef}

\begin{mydef}
	$$ \sigma_{\Sigma}(B) = \llbracket y^n | n = \sum_{x \in \{ z | z \in \{B\} \land \sigma_{\Sigma}(z) = y \} } B(x) \rrbracket $$
\end{mydef}
